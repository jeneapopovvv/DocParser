def preprocess_document_image2(
    image_path,
    min_short_side=800,
    max_long_side=1600,
    target_max_size_bytes=1_200_000,  # 1.2 MB
    initial_quality=95,
    min_quality=70,
):
    """
    Preprocess a document image:
    - Enhance contrast
    - Resize within bounds (no upscaling)
    - Ensure final file size <= target_max_size_bytes
    Returns: BytesIO object (JPEG format)
    """
    img = Image.open(image_path).convert("RGB")
    
    # Step 1: Enhance contrast
    img = ImageOps.autocontrast(img, cutoff=2)
    
    # Step 2: Resize within bounds (do not upscale)
    w, h = img.size
    scale = min(
        max_long_side / max(w, h),
        min_short_side / min(w, h),
        1.0  # no upscaling
    )
    if scale < 1.0:
        new_w, new_h = int(w * scale), int(h * scale)
        img = img.resize((new_w, new_h), Image.Resampling.LANCZOS)
    
    # Step 3: Optimize file size to <= target_max_size_bytes
    buffer = BytesIO()
    
    # Try initial high-quality save
    img.save(buffer, format="JPEG", quality=initial_quality, optimize=True)
    
    if buffer.tell() <= target_max_size_bytes:
        buffer.seek(0)
        return buffer
    
    # If too large, reduce quality iteratively
    quality = initial_quality
    while quality >= min_quality:
        buffer = BytesIO()
        img.save(buffer, format="JPEG", quality=quality, optimize=True)
        if buffer.tell() <= target_max_size_bytes:
            buffer.seek(0)
            return buffer
        quality -= 5
    
    # If still too large after min_quality, downscale further
    current_w, current_h = img.size
    scale_factor = (target_max_size_bytes / buffer.tell()) ** 0.5  # approx. linear in area
    scale_factor = max(scale_factor, 0.5)  # avoid extreme downscaling
    
    new_w = int(current_w * scale_factor)
    new_h = int(current_h * scale_factor)
    if new_w < 400 or new_h < 400:
        # Safety: don't go below minimal legibility
        new_w, new_h = max(new_w, 400), max(new_h, 400)
    
    img = img.resize((new_w, new_h), Image.Resampling.LANCZOS)
    
    # Final attempt with min_quality
    buffer = BytesIO()
    img.save(buffer, format="JPEG", quality=min_quality, optimize=True)
    
    # If still over (rare), force truncate (not ideal, but meets spec)
    if buffer.tell() > target_max_size_bytes:
        # Re-encode at even lower quality as last resort
        extra_low_q = max(50, min_quality - 10)
        buffer = BytesIO()
        img.save(buffer, format="JPEG", quality=extra_low_q, optimize=True)
    
    buffer.seek(0)
    return buffer




def preprocess_document_image(image_path, min_short_side=800, max_long_side=1600):
    img = Image.open(image_path).convert("RGB")
    
    # Auto-contrast for better text clarity
    img = ImageOps.autocontrast(img, cutoff=2)
    
    # Resize while preserving aspect ratio
    w, h = img.size
    scale = min(
        max_long_side / max(w, h),
        min_short_side / min(w, h),
        1.0  # don't upscale
    )
    new_w, new_h = int(w * scale), int(h * scale)
    img = img.resize((new_w, new_h), Image.Resampling.LANCZOS)
    
    return img




def build_request(images: list[Image.Image], question: str) -> dict:
    """Build a single vLLM request dict for N images + a question."""
    messages = [{
        "role": "user",
        "content": [
            *[{"type": "image", "image": img} for img in images],
            {"type": "text", "text": question},
        ],
    }]
 
    prompt = processor.apply_chat_template(
        messages,
        tokenize=False,
        add_generation_prompt=True,
        enable_thinking=False
    )
 
    return {
        "prompt": prompt,
        "multi_modal_data": {"image": images},
    }
